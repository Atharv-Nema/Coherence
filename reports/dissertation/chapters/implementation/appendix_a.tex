\section{\textit{Algorithm pseudocode}}
\subsection{Return checker}
\begin{lstlisting}{language=ocaml}
type status = Returns | DoesNotReturn | Invalid

(* Merges paths for branching logic *)
let merge s1 s2 = 
    match (s1, s2) with
    | (Invalid, _) | (_, Invalid) -> Invalid
    | (Returns, Returns)          -> Returns
    | _                           -> DoesNotReturn

(* Evaluates a single statement *)
let rec stmt_return_status stmt = match stmt with
    | Return _    -> Returns
    | Atomic body -> stmt_list_return_status body
    | If (_, then_b, Some else_b) -> 
        let s1 = stmt_return_status (last then_b) in
        let s2 = stmt_return_status (last else_b) in
        merge s1 s2
    | While (_, body) -> 
        (* Loops are never guaranteed to return as the 
           condition may be false *)
        merge (stmt_list_return_status body) DoesNotReturn
    | _ -> DoesNotReturn

(* Evaluates a sequence, detecting dead code *)
and stmt_list_return_status stmts = match stmts with
    | [] -> DoesNotReturn
    | [s] -> stmt_return_status s
    | s :: rest -> 
        match stmt_return_status s with
        (* Error: Dead code follows a return *)
        | Returns | Invalid -> Invalid 
        | DoesNotReturn -> stmt_list_return_status rest
\end{lstlisting}
